### 2.3 Relational Database Schemas

```sql
-- Mission Control Schema
CREATE TABLE business_missions (
    id UUID PRIMARY KEY,
    mission_name VARCHAR(255) NOT NULL,
    objective TEXT,
    kpi_targets JSONB,
    status ENUM('planning', 'active', 'completed', 'paused'),
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE mission_workflows (
    id UUID PRIMARY KEY,
    mission_id UUID REFERENCES business_missions(id),
    workflow_type VARCHAR(100),
    trigger_conditions JSONB,
    execution_steps JSONB,
    automation_level DECIMAL(3,2), -- 0.00 to 1.00
    last_executed TIMESTAMP
);

CREATE TABLE revenue_events (
    id UUID PRIMARY KEY,
    timestamp TIMESTAMP NOT NULL,
    amount DECIMAL(12,2),
    currency VARCHAR(3),
    source VARCHAR(100),
    channel VARCHAR(50),
    sku VARCHAR(50),
    order_id VARCHAR(100),
    kind ENUM('real', 'simulated', 'test'),
    asset_url TEXT
);

CREATE TABLE kpi_measurements (
    id UUID PRIMARY KEY,
    kpi_id VARCHAR(50),
    measured_at TIMESTAMP,
    actual_value DECIMAL(12,4),
    target_value DECIMAL(12,4),
    status ENUM('green', 'yellow', 'red'),
    notes TEXT
);
```
